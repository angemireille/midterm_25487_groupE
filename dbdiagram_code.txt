// Copy this code to dbdiagram.io

Table locations {
  id UUID [pk]
  code varchar(50) [unique]
  name varchar(100)
  type varchar(20) // PROVINCE, DISTRICT, SECTOR, CELL, VILLAGE
  parent_id UUID [ref: > locations.id]
}

Table member {
  member_id int [pk]
  name varchar(45)
  email varchar(45)
  password varchar(255)
  role varchar(20)
  reset_token varchar(64)
  reset_token_expiry timestamp
  two_factor_code varchar(6)
  two_factor_expiry timestamp
  village_id UUID [ref: > locations.id]
  address_id int [ref: - address.id]
}

Table address {
  id int [pk]
  street varchar(100)
  city varchar(50)
  province varchar(50)
  postalCode varchar(20)
  country varchar(50)
}

Table category {
  category_id int [pk]
  name varchar(45)
}

Table book {
  book_id int [pk]
  book_title varchar(45)
  language varchar(30)
  total_copies int
  available_copies int
  category_id int [ref: > category.category_id]
}

Table author {
  author_id int [pk]
  name varchar(45)
}

Ref: book.book_id <> author.author_id // Many-to-Many

Table borrow {
  borrowId int [pk]
  member_id int [ref: > member.member_id]
  book_id int [ref: > book.book_id]
  borrowDate timestamp
  dueDate timestamp
  returnDate timestamp
  fine decimal
}
